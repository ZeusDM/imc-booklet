\begin{itemize}
    \item[(a)] Prove que
    \[ \lim_{x\to \infty}\,\sum_{n=1}^{\infty} \frac{nx}{(n^{2}+x)^{2}}=\frac{1}{2}.\]
    \item[(b)] Prove que existe uma constante positiva \(c\) tal que para todo \(x\in [1,\infty)\) temos
    \[\left|\sum_{n=1}^{\infty} \frac{nx}{(n^{2}+x)^{2}}-\frac{1}{2} \right| \leq \frac{c}{x}.\]
\end{itemize}
