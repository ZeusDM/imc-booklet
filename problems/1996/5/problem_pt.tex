\begin{itemize}
    \item[(a)] Seja \(a,b\) números reais tais que \(b\leq 0\) e \(1+ax+bx^{2} \geq 0\) para todo \(x\in [0,1]\).
    Prove que
    \[\lim_{n\to \infty} n \int_{0}^{1}(1+ax+bx^{2})^{n}dx= \begin{cases}
    -\frac{1}{a} &\text{se}\; a<0,\\
    \infty & \text{se}\; a \geq 0.
    \end{cases}\]
    \item[(b)] Seja \(f\colon [0,1]\rightarrow[0,\infty)\) uma função com uma segunda derivada contínua e \(f''(x)\leq0\) para todo \(x\in [0,1]\). Suponha que \(L=\lim_{n\to \infty} n \int_{0}^{1}(f(x))^{n}dx\) existe e \(0<L<\infty\). Prove que \(f'\) tem sinal constante e \(\min_{x\in [0,1]}|f'(x)|=L^{-1}\).
\end{itemize}
