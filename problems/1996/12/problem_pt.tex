\begin{itemize}
\item[(a)] Prove que para toda sequência \((a_{n})_{n\in \mathbb{N}}\), tal que \(a_{n}>0\) para todo \(n \in \mathbb{N}\) e
\(\sum_{n=1}^{\infty}a_{n}<\infty\), temos
\[\sum_{n=1}^{\infty}(a_{1}a_{2} \cdots a_{n})^{\frac{1}{n}}< e\sum_{n=1}^{\infty}a_{n}.\]
\item[(b)] Prove que para todo \(\epsilon>0\) existe uma sequência \((b_{n})_{n\in \mathbb{N}}\) tal que \(b_{n}>0\) para todo \(n \in \mathbb{N}\) e
\(\sum_{n=1}^{\infty}b_{n}<\infty\) e
\[\sum_{n=1}^{\infty}(b_{1}b_{2} \cdots b_{n})^{\frac{1}{n}}> (e-\epsilon)\sum_{n=1}^{\infty}b_{n}.\]
\end{itemize}
