\begin{itemize}
\item[(a)] Seja \(f\colon  \mathbb{R}^{n\times n}\to\mathbb{R}\) uma aplicação linear. Prove que \(\exists ! C\in\mathbb{R}^{n\times n}\) tal que \(f(A)=Tr(AC), \forall A \in \mathbb{R}^{n\times n}\).

\item[(b)] Suponha além disso que \(\forall A,B \in \mathbb{R}^{n\times n}: f(AB)=f(BA)\). Prove que \(\exists \lambda \in \mathbb{R}: f(A)=\lambda Tr(A)\).
\end{itemize}
