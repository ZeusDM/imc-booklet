\begin{itemize}
\item[(a)]  Mostre que para cada função \(f\colon \mathbb{Q} \times \mathbb{Q} \to \mathbb{R}\), existe uma função \(g\colon \mathbb{Q}\to \mathbb{R}\) com \(f(x,y) \leq g(x)+g(y) \) para todos \(x,y\in \mathbb{Q}\).
\item[(b)] Determine uma função \(f\colon \mathbb{R} \times \mathbb{R} \to \mathbb{R}\), para a qual não existe uma função \(g\colon \mathbb{Q}\to \mathbb{R}\) tal que \(f(x,y) \leq g(x)+g(y) \) para todos \(x,y\in \mathbb{R}\).
\end{itemize}
