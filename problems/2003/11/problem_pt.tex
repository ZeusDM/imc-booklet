\begin{itemize}
\item[(a)]  Mostre que para cada função \(f:\mathbb{Q} \times \mathbb{Q} \rightarrow \mathbb{R}\), existe uma função \(g:\mathbb{Q}\rightarrow \mathbb{R}\) com \(f(x,y) \leq g(x)+g(y) \) para todo \(x,y\in \mathbb{Q}\).
\item[(b)] Encontre uma função \(f:\mathbb{R} \times \mathbb{R} \rightarrow \mathbb{R}\), para a qual não existe uma função \(g:\mathbb{Q}\rightarrow \mathbb{R}\) tal que \(f(x,y) \leq g(x)+g(y) \) para todo \(x,y\in \mathbb{R}\).
\end{itemize}
