\begin{itemize}
\item[(a)]  Show that for each function \(f\colon \mathbb{Q} \times \mathbb{Q} \to \mathbb{R}\), there exists a function \(g\colon \mathbb{Q}\to \mathbb{R}\) with \(f(x,y) \leq g(x)+g(y) \) for all \(x,y\in \mathbb{Q}\).
\item[(b)] Find a function \(f\colon \mathbb{R} \times \mathbb{R} \to \mathbb{R}\), for which there is no function \(g\colon \mathbb{Q}\to \mathbb{R}\) such that \(f(x,y) \leq g(x)+g(y) \) for all \(x,y\in \mathbb{R}\).
\end{itemize}
