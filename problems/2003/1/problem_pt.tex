\begin{itemize}
\item[(a)] Seja \(a_1,a_2,\dots\) uma sequência de números reais com \(a_1=1\) e \(a_{n+1}>\frac32 a_n\) para todo \(n\). Prove que \(\lim_{n\rightarrow\infty}\frac{a_n}{\left(\frac32\right)^{n-1}}\) existe. (finito ou infinito)

\item[(b)] Prove que para todo \(\alpha>1\) existe uma sequência \(a_1,a_2,\dots\) com as mesmas propriedades tal que \(\lim_{n\rightarrow\infty}\frac{a_n}{\left(\frac32\right)^{n-1}}=\alpha\)
\end{itemize}
