\begin{itemize}
\item[(a)] Prove que para todo \(\epsilon>0\) existe um número inteiro positivo \(n\) e números reais \(\lambda_{1},\dots,\lambda_{n}\) tais que
\[\max_{x\in [-1,1]}|x-\sum_{k=1}^{n}\lambda_{k}x^{2k+1}|<\epsilon.\]
\item[(b)] Prove que para toda função contínua ímpar \(f\) em \([-1,1]\) e para todo \(\epsilon>0\) existe um número inteiro positivo \(n\) e números reais \(\mu_{1},\dots,\mu_{n}\) tais que
\[\max_{x\in [-1,1]}|f(x)-\sum_{k=1}^{n}\mu_{k}x^{2k+1}|<\epsilon.\]
\end{itemize}
