\begin{itemize}
\item[(a)] Prove que toda função da forma
\[f(x)=\frac{a_{0}}{2}+\cos(x)+\sum_{n=2}^{N}a_{n}\cos(nx)\]com \(|a_{0}|<1\) possui valores positivos e negativos no período \([0,2\pi)\).
\item[(b)] Prove que a função
\[F(x)=\sum_{n=1}^{100}\cos(n^{\frac{3}{2}}x)\]tem pelo menos \(40\) zeros no intervalo \((0,1000)\).
\end{itemize}
