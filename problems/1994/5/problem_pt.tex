\begin{itemize}
\item[(a)] Seja \(f\in C[0,b]\), \(g\in C(\mathbb R)\) e seja \(g\) periódica com período \(b\). Prove que \(\int_0^b f(x) g(nx)\,\mathrm dx\) possui limite quando \(n\to\infty\) e
\[\lim_{n\to\infty}\int_0^b f(x)g(nx)\,\mathrm dx=\frac 1b \int_0^b f(x)\,\mathrm dx\cdot\int_0^b g(x)\,\mathrm dx\]
\item[(b)] Calcule
\[\lim_{n\to\infty}\int_0^\pi \frac{\sin x}{1+3\cos^2nx}\,\mathrm dx\]
\end{itemize}
